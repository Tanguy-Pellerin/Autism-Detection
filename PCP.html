for subdir, dirs, files in os.walk("csv_clean/"):
    for file in files:
        #print os.path.join(subdir, file)
        filepath = subdir + "/" + file

        if filepath.endswith(".csv"):
            data = genfromtxt(filepath, delimiter=',')
            img = Image.fromarray(data, 'RGB')
            newsize = (32, 156) 
            im1 = img.resize(newsize)
            im1.save('data_image/' + filepath[10:-3] + 'jpg')

train["categorie"] = 1

train["categorie"] = train["categorie"].apply(str)


train_datagen = ImageDataGenerator(
    rescale=1./255,
    shear_range=0.1,
    zoom_range=0.2,
    width_shift_range=0.1,
    height_shift_range=0.1
)

train_generator = train_datagen.flow_from_dataframe(
    train, 
    "data_image/autism/", 
    x_col='image',
    y_col='categorie',
    target_size=(156, 32),
    class_mode='categorical',
    batch_size=152
)

def save_image(data, fn):
   
    sizes = np.shape(data)
    height = float(sizes[0])
    width = float(sizes[1])
     
    fig = plt.figure()
    fig.set_size_inches(width/height, 1, forward=False)
    ax = plt.Axes(fig, [0., 0., 1., 1.])
    ax.set_axis_off()
    fig.add_axes(ax)
 
    ax.imshow(data)
    plt.savefig(fn, dpi = height) 
    plt.close()

name=0
for a in tqdm(range(len(train))):
    for i in range(0, 153):
        for X_batch, Y_batch in train_generator:
            image = X_batch[0]
            plt.imshow(image)
            plt.axis('off')
            save_image(image,'augmented_image/autism/'+str(name)+'.jpg')
            name+=1
            break
